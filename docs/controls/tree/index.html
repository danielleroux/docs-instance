@Component({

    <ix-tree root="root" [model]="model"></ix-tree>

export class Tree {
  model: TreeModel<{
    name: string;
  }> = {
    root: {
      id: 'root',
      data: {
        name: '',
      },

      children: ['sample'],
    },
    sample: {
      id: 'sample',
      data: {
        name: 'Sample',
      },

    'sample-child-1': {
      id: 'sample-child-1',
      data: {
        name: 'Sample Child 1',
      },
      hasChildren: false,
      children: [],
    },
    'sample-child-2': {
      id: 'sample-child-2',
      data: {
        name: 'Sample Child 2',
      },
      hasChildren: false,
      children: [],
    },
  };
}

import { TreeModel } from '@siemens/ix';
import { IxTree } from '@siemens/ix-react';
import React, { useState } from 'react';

type TreeData = {
  name: string;
  icon: string;
};

export const Tree: React.FC = () => {
  const [model] = useState<TreeModel<TreeData>>({
    root: {
      id: 'root',
      data: {
        icon: '',
        name: '',
      },
      hasChildren: true,
      children: ['sample'],
    },
    sample: {
      id: 'sample',
      data: {
        name: 'Sample',
        icon: 'star',
      },
      hasChildren: true,
      children: ['sample-child-1', 'sample-child-2'],
    },
    'sample-child-1': {
      id: 'sample-child-1',
      data: {
        name: 'Sample Child 1',
        icon: 'cut',
      },
      hasChildren: false,
      children: [],
    },
    'sample-child-2': {
      id: 'sample-child-2',
      data: {
        name: 'Sample Child 2',
        icon: 'print',
      },
      hasChildren: false,
      children: [],
    },
  });
  return (
    <div
      style={{
        display: 'block',
        position: 'relative',
        width: '100%',
        height: '40rem',
      }}
    >
      <IxTree root="root" model={model}></IxTree>
    </div>
  );
};

  model: TreeModel<TreeData> = {

import { TreeContext, TreeModel } from '@siemens/ix';
import { IxButton, IxIcon, IxTree } from '@siemens/ix-react';
import React, { useState } from 'react';

type TreeData = {
  name: string;
  icon: string;
};

export const TreeCustom: React.FC = () => {
  const [context, setContext] = useState<TreeContext>();
  const [model] = useState<TreeModel<TreeData>>({
    root: {
      id: 'root',
      data: {
        icon: '',
        name: '',
      },
      hasChildren: true,
      children: ['sample'],
    },
    sample: {
      id: 'sample',
      data: {
        name: 'Sample',
        icon: 'star',
      },
      hasChildren: true,
      children: ['sample-child-1', 'sample-child-2'],
    },
    'sample-child-1': {
      id: 'sample-child-1',
      data: {
        name: 'Sample Child 1',
        icon: 'cut',
      },
      hasChildren: false,
      children: [],
    },
    'sample-child-2': {
      id: 'sample-child-2',
      data: {
        name: 'Sample Child 2',
        icon: 'print',
      },
      hasChildren: false,
      children: [],
    },
  });

  function expandAndSelect() {
    setContext({
      sample: {
        isExpanded: true,
        isSelected: false,
      },
      'sample-child-2': {
        isSelected: true,
        isExpanded: false,
      },
    });
  }

  return (
    <div
      style={{
        display: 'block',
        position: 'relative',
        width: '100%',
        height: '40rem',
      }}
    >
      <IxButton
        onClick={expandAndSelect}
        ghost
        style={{ marginBottom: '2rem' }}
      >
        Expand Tree
      </IxButton>
      <IxTree
        root="root"
        model={model}
        context={context}
        onContextChange={({ detail }) => {
          setContext(detail);
        }}
        renderItem={(data: TreeData) => (
          <div className="d-flex align-items-center">
            <IxIcon name={data.icon} size="16" className="me-2" /> {data.name}
          </div>
        )}
      ></IxTree>
    </div>
  );
};

<div style="height: 8rem; width: 100%">
  <ix-button id="expand" ghost style="margin-bottom: 2rem"
    >Expand Tree</ix-button
  >
  <ix-tree root="root" id="tree"></ix-tree>
</div>

<script type="text/javascript">
  (async function () {
    await window.customElements.whenDefined('ix-tree');
    const tree = document.getElementById('tree');

    const expandButton = document.getElementById('expand');
    expandButton.addEventListener('click', () => {
      tree.context = {
        sample: {
          isExpanded: true,
          isSelected: false,
        },
        'sample-child-2': {
          isSelected: true,
          isExpanded: false,
        },
      };
    });

    tree.renderItem = (index, item, dataList, context, update) => {
      const el = document.createElement('ix-tree-item');
      el.hasChildren = item.hasChildren;
      el.context = context[item.id];

      const div = document.createElement('div');
      div.style.display = 'flex';

      const name = document.createElement('span');
      const icon = document.createElement('ix-icon');
      icon.name = item.data.icon;
      icon.style.marginRight = '0.5rem';

      div.appendChild(icon);
      div.appendChild(name);

      name.innerText = item.data.name;

      el.appendChild(div);

      update((updateTreeItem) => {
        name.innerText = updateTreeItem.data.name;
      });

      return el;
    };

    tree.model = {
      root: {
        id: 'root',
        data: null,
        hasChildren: true,
        children: ['sample'],
      },
      sample: {
        id: 'sample',
        data: {
          name: 'Sample',
          icon: 'star',
        },
        hasChildren: true,
        children: ['sample-child-1', 'sample-child-2'],
      },
      'sample-child-1': {
        id: 'sample-child-1',
        data: {
          name: 'Sample Child 1',
          icon: 'cut',
        },
        hasChildren: false,
        children: [],
      },
      'sample-child-2': {
        id: 'sample-child-2',
        data: {
          name: 'Sample Child 2',
          icon: 'print',
        },
        hasChildren: false,
        children: [],
      },
    };
  })();
</script>
